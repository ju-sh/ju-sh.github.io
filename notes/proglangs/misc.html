C++

vector

  • Default value of elements of a vector<T> is 0
#include<iostream>
#include<vector>

int main() {
    std::vector<int> v1 = {1, 2, 3};
    std::cout << v1[0] << std::endl;  // 1

    std::vector<int> v2(3);
    std::cout << v2[0] << std::endl;  // 0

    std::vector<bool> v3(3);
    std::cout << v3[0] << std::endl;  // 0
}
